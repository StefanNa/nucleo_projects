115 int main(void)
116 {
117  int n_menu_state=MAIN_MENU;
118  uint8_t uch_first_entry = 1;
119  uint8_t uch_input;
120  uint16_t u_miso;
121  uint8_t u_byte = 0;
122 
123  maxim_gpio_init();
124  GPIO_ResetBits(GPIOA, GPIO_Pin_0); // LED_G = 0.
125  GPIO_ResetBits(GPIOA, GPIO_Pin_15); // LED_R = 0.
126  maxim_spi_init();
127  maxim_uart_init();
128 
129  // Clear the Screen, and then display the big Maxim banner for about 2 seconds
130  maxim_menu_cls();
131  maxim_menu_print_maxim_banner_big();
132  printf("Maxim MAXREFDES64#\r\n");
133  printf("Revision v%d.%d\r\n",MAJOR_REVISION,MINOR_REVISION);
134 
135  maxim_delay(2*ONE_SECOND); //2 second pause
136 
137  while(1)
138  {
139  switch(n_menu_state)
140  {
141  case MAIN_MENU:
142  maxim_menu_cls();
143  maxim_menu_print_main_menu();
144  do{
145  if(uch_first_entry)
146  {
147  if(USART_GetFlagStatus(USART1,USART_FLAG_RXNE)==RESET)
148  {
149  maxim_delay(ONE_SECOND); //1 sec pause
150  }
151  if(USART_GetFlagStatus(USART1,USART_FLAG_RXNE)==RESET)
152  {
153  maxim_delay(ONE_SECOND); //1 sec pause
154  n_menu_state = MAIN_MENU;
155  break;
156  }
157  }
158 
159  uch_input = getchar();
160 
161  if(uch_input == '0')
162  n_menu_state = CONTINUOUS_READ_MENU;
163  else if(uch_input == '1')
164  n_menu_state = SINGLE_READ_MENU;
165 
166  uch_first_entry=0;
167  printf("%c\r\n",uch_input);
168  break;
169  }while(1);
170  break;
171 
172  case CONTINUOUS_READ_MENU:
173  printf("Press ESC to stop Continuous Read Mode\r\n");
174  do{
175  if(USART_GetFlagStatus(USART1,USART_FLAG_RXNE)==SET)
176  {
177  uch_input = getchar();
178 
179  if(uch_input == 0x1B)
180  {
181  n_menu_state = MAIN_MENU;
182  break;
183  }
184  }
185  maxim_delay(ONE_SECOND);
186  u_miso = maxim_spi_16bit_transfer(0x00);
187  u_byte = u_miso>>8;
188  printf("0X%02X <", u_byte);
189  printf(BYTETOBINARYPATTERN, BYTETOBINARY(u_byte));
190  printf(">\r\n");
191  printf("Press ESC to stop Continuous Read Mode\r\n");
192  }while(1);
193  break;
194 
195  case SINGLE_READ_MENU:
196  printf("Press SPACE BAR to Read the Input\r\n");
197  printf("Press ESC to stop Single Read Mode\r\n");
198  do{
199  if(USART_GetFlagStatus(USART1,USART_FLAG_RXNE)==SET)
200  {
201  uch_input = getchar();
202 
203  if(uch_input == 0x1B)
204  {
205  n_menu_state = MAIN_MENU;
206  break;
207  }
208  else if(uch_input == 0x20)
209  {
210  u_miso = maxim_spi_16bit_transfer(0x00);
211  u_byte = u_miso>>8;
212  printf("0X%02X <", u_byte);
213  printf(BYTETOBINARYPATTERN, BYTETOBINARY(u_byte));
214  printf(">\r\n");
215  printf("Press SPACE BAR to Read the Input\r\n");
216  printf("Press ESC to stop Single Read Mode\r\n");
217  }
218  }
219  }while(1);
220 
221  break;
222 
223  default:
224  n_menu_state = MAIN_MENU;
225  break;
226  }
227  }
228 }
